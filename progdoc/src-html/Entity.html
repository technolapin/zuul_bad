001/**
002 * classe mere de Room et Player
003 * @author Cl√©ment Chomicki
004 */
005public abstract class Entity
006{
007    private String    aName;
008    private Inventory aInventory;
009
010    /** Constructeur de la classe
011     * @param pN le nom de l'entite (sa description courte)
012     */
013    public Entity( final String pN )
014    {
015        this.aName      = pN;
016        this.aInventory = new Inventory();
017    } //Entity()
018
019    /**
020     * Accesseur de aName
021     * @return le nom de l'entite
022     */
023    public String getName()
024    {
025        return this.aName;
026    } //getName()
027    
028    /**
029     * Cherche un item dans l'inventaire de l'entite via son nom
030     * @param pItemName le nom de l'item cherche
031     * @return un item ou null
032     */
033    public Item getItemByName( final String pItemName )
034    {
035        return this.aInventory.getByName( pItemName );
036    } //getItemByName()
037    
038    /**
039     * Verifie la presence d'un item dans l'inventaire
040     * @param pItem l'item cherche
041     * @return un booleen traduisant la presence de l'item
042     */
043    public boolean hasItem( final Item pItem )
044    {
045        return this.aInventory.hasItem( pItem );
046    } //hasItem()
047
048    /**
049     * Verifie si l'inventaire contient au moins un item
050     * @return un booleen
051     */
052    public boolean hasItems()
053    {
054        return (! this.aInventory.isEmpty());
055    } //hasItems()
056    
057    /**
058     * ajoute un item a l'inventaire
059     * @param pItem l'item a ajouter
060     */
061    public void addItem( final Item pItem )
062    {
063        this.aInventory.add( pItem );
064    } //addItem()
065
066    /**
067     * ajoute des items a l'inventaire
068     * @param pItem l'item a ajouter
069     * @param pNombre le nombre de fois que pItem doit etre ajoute
070     */
071    public void addItem( final Item pItem, final int pNombre )
072    {
073        this.aInventory.add( pItem, pNombre);
074    } //addItem()
075
076    /**
077     * retire un item de l'inventaire
078     * @param pItem l'item a supprimer
079     */
080    public void removeItem( final Item pItem )
081    {
082        this.aInventory.remove( pItem );
083    } //removeItem()
084    
085    /**
086     * Retourne une chaine decrivant le contenu de l'inventaire
087     * @return une chaine decrivant le contenu de l'inventaire
088     */
089    public String inventoryString()
090    {
091        return this.aInventory.toString();
092    } //inventoryString()
093 
094    /**
095     * retourne le poids total
096     *@return le poids total
097     */
098    public int getCarriedWeight()
099    {
100        return this.aInventory.totalWeight();
101    } //getCarriedWeight()
102
103}
