002 * classe m√®re de Room et Player
003 */
004public abstract class Entity
005{
006    private String    aName;
007    private Inventory aInventory;
008
009    /** Constructeur de la classe
010     * @param pN le nom de l'entite (sa description courte)
011     */
012    public Entity( final String pN )
013    {
014        this.aName      = pN;
015        this.aInventory = new Inventory();
016    } //Entity()
017
018    /**
019     * Accesseur de aName
020     * @return le nom de l'entite
021     */
022    public String getName()
023    {
024        return this.aName;
025    } //getName()
026    
027    /**
028     * Cherche un item dans l'inventaire de l'entite via son nom
029     * @param pItemName le nom de l'item cherche
030     * @return un item ou null
031     */
032    public Item getItemByName( final String pItemName )
033    {
034        return this.aInventory.getByName( pItemName );
035    } //getItemByName()
036    
037    /**
038     * Verifie la presence d'un item dans l'inventaire
039     * @param pItem l'item cherche
040     * @return un booleen traduisant la presence de l'item
041     */
042    public boolean hasItem( final Item pItem )
043    {
044        return this.aInventory.hasItem( pItem );
045    } //hasItem()
046
047    /**
048     * Verifie si l'inventaire contient au moins un item
049     * @return un booleen
050     */
051    public boolean hasItems()
052    {
053        return (! this.aInventory.isEmpty());
054    } //hasItems()
055    
056    /**
057     * ajoute un item a l'inventaire
058     * @param pItem l'item a ajouter
059     */
060    public void addItem( final Item pItem )
061    {
062        this.aInventory.add( pItem );
063    } //addItem()
064
065    /**
066     * ajoute des items a l'inventaire
067     * @param pItem l'item a ajouter
068     * @param pNombre le nombre de fois que pItem doit etre ajoute
069     */
070    public void addItem( final Item pItem, final int pNombre )
071    {
072        this.aInventory.add( pItem, pNombre);
073    } //addItem()
074
075    /**
076     * retire un item de l'inventaire
077     * @param pItem l'item a supprimer
078     */
079    public void removeItem( final Item pItem )
080    {
081        this.aInventory.remove( pItem );
082    } //removeItem()
083    
084    /**
085     * Retourne une chaine decrivant le contenu de l'inventaire
086     * @return une chaine decrivant le contenu de l'inventaire
087     */
088    public String inventoryString()
089    {
090        return this.aInventory.toString();
091    } //inventoryString()
092 
093    /**
094     * retourne le poids total
095     *@return le poids total
096     */
097    public int getCarriedWeight()
098    {
099        return this.aInventory.totalWeight();
100    } //getCarriedWeight()
101
102}
