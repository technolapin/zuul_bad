001/**
002 * Item permettant de se téléporter
003 * @author Clément Chomicki
004 */
005public class Beamer extends Item
006{
007    private Room aDestination;
008
009
010    /**
011     * constructeur de la classe
012     * @param pDescr la description de l'item
013     * @param pWei le poids de l'item
014     */
015    public Beamer(final String pDescr, final int pWei)
016    {
017        super(pDescr, pWei);
018    } //Beamer()
019
020
021    /**
022     * Teste si le beamer est charge
023     * @return true si le beamer est charge
024     */
025    public boolean isCharged()
026    {
027        return this.aDestination != null;
028    } //isCharged()
029
030    
031    /**
032     * charge le beamer
033     * @param pRoom la piece dans laquelle on charge le beamer
034     */
035    public void charge(final Room pRoom)
036    {
037        this.aDestination = pRoom;
038    } //charge()
039
040    
041    /**
042     * decharge le beamer (lors de son utilisation pour se teleporter)
043     * @return la piece stockee dans le beamer
044     */
045    public Room discharge()
046    {
047        Room vRoom = this.aDestination;
048        this.aDestination = null;
049        return vRoom;
050    } //discharge()
051
052}
