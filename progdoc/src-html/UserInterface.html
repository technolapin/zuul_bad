026/**
027 * This class implements a simple graphical user interface with a text entry
028 * area, a text output area and an optional image.
029 * 
030 * @author Michael Kolling (DB edited)
031 * @author Clément Chomicki
032 * @version 1.0 (Jan 2003)
033 */
034public class UserInterface implements ActionListener
035{
036    private GameEngine aEngine;
037    private JFrame     aMyFrame;
038    private JTextField aEntryField;
039    private JTextArea  aLog;
040    private JLabel     aImage;
041
042    /**
043     * Construct a UserInterface. As a parameter, a Game Engine
044     * (an object processing and executing the game commands) is
045     * needed.
046     * 
047     * @param pGameEngine  The GameEngine object implementing the game logic.
048     */
049    public UserInterface( final GameEngine pGameEngine )
050    {
051        this.aEngine = pGameEngine;
052        this.createGUI();
053    } // UserInterface(.)
054
055    /**
056     * Print out some text into the text area.
057     * @param pText le texte a afficher
058     */
059    public void print( final String pText )
060    {
061        this.aLog.append( pText );
062        this.aLog.setCaretPosition( this.aLog.getDocument().getLength() );
063    } // print(.)
064
065    /**
066     * Print out some text into the text area, followed by a line break.
067     * @param pText le text a afficher
068     */
069    public void println( final String pText )
070    {
071        this.print( pText + "\n" );
072    } // println(.)
073
074    /**
075     * Show an image file in the interface.
076     * @param pImageName le nom de l'image a afficher.
077     */
078    public void showImage( final String pImageName )
079    {
080        URL vImageURL = this.getClass().getClassLoader().getResource( pImageName );
081        if ( vImageURL == null )
082            System.out.println( "image not found" );
083        else {
084            ImageIcon vIcon = new ImageIcon( vImageURL );
085            this.aImage.setIcon( vIcon );
086            this.aMyFrame.pack();
087        }
088    } // showImage(.)
089
090    /**
091     * Enable or disable input in the input field.
092     * @param pOnOff un booleen
093     */
094    public void enable( final boolean pOnOff )
095    {
096        this.aEntryField.setEditable( pOnOff );
097        if ( ! pOnOff )
098            this.aEntryField.getCaret().setBlinkRate( 0 );
099    } // enable(.)
100
101    /**
102     * Set up graphical user interface.
103     */
104    private void createGUI()
105    {
106        this.aMyFrame = new JFrame( "Zork" );
107        
108        this.aEntryField = new JTextField( 34 );
109        this.aEntryField.setBackground(new Color(10, 23, 0));
110        this.aEntryField.setForeground(new Color(150, 180, 100));
111        //        this.aLog = new JTextArea();
112        this.aLog = new FancyTextArea(RessourcesFetcher.openImage("background_text"));
113        /*
114          this.aLog.setImage(
115          RessourcesFetcher.openImage("corgy")
116          );
117        */
118        this.aLog.setEditable( false );
119
120        JScrollPane vListScroller = new JScrollPane( this.aLog );
121        vListScroller.setPreferredSize( new Dimension(200, 200) );
122        vListScroller.setMinimumSize( new Dimension(100,100) );
123
124        JPanel vPanel = new JPanel();
125        this.aImage = new JLabel();
126
127        vPanel.setLayout( new BorderLayout() );
128
129        vPanel.add( this.aImage, BorderLayout.NORTH );
130        vPanel.add( vListScroller, BorderLayout.CENTER );
131        vPanel.add( this.aEntryField, BorderLayout.SOUTH );
132
133        this.aMyFrame.getContentPane().add( vPanel, BorderLayout.CENTER );
134
135        // add some event listeners to some components
136        this.aMyFrame.addWindowListener( new WindowAdapter() {
137            public void windowClosing(WindowEvent e) { System.exit(0); }
138        } );
139
140        this.aEntryField.addActionListener( this );
141
142        this.aMyFrame.pack();
143        this.aMyFrame.setVisible( true );
144        this.aEntryField.requestFocus();
145    } // createGUI()
146
147    /**
148     * Actionlistener interface for entry textfield.
149     * @param pE un ActionEvent
150     */
151    public void
152        actionPerformed( final ActionEvent pE ) 
153    {
154        String vActionCmd =  pE.getActionCommand();
155        if (vActionCmd=="El famoso")
156            this.println("!!VOUS AVEZ CLIQUÉ SUR EL FAMOSO!!!");
157        else
158            this.processCommand();
159    } // actionPerformed(.)
160
161    
162    /**
163     * A command has been entered. Read the command and do whatever is 
164     * necessary to process it.
165     */
166    private void processCommand()
167    {
168        String vInput = this.aEntryField.getText();
169        this.aEntryField.setText( "" );
170
171        this.aEngine.interpretCommand( vInput );
172    } // processCommand()
173    
174} // UserInterface
