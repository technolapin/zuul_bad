217    /**
218     * Methode traitant une commande.
219     * @param pCommandLine: String entree
220     */
221    public void interpretCommand (final String pCommandLine)
222    {
223        this.aGui.println(pCommandLine);
224        Command vCommand = this.aParser.getCommand(pCommandLine);
225
226        if (vCommand.isUnknown())
227            {
228                this.aGui.println ( "I don't know what you mean..." );
229            }
230        else
231            switch (vCommand.getCommandWord())
232                {
233                case GO   :
234                    this.goRoom(vCommand);
235                    break;
236                    
237                case HELP :
238                    this.printHelp();
239                    break;
240
241                case QUIT :
242                    if ( vCommand.hasSecondWord() )
243                        aGui.println("Quit what?");
244                    else
245                        this.endGame();
246                    break;
247                    
248                case LOOK :
249                    this.look();
250                    break;
251
252                case EAT  :
253                    this.eat(vCommand);
254                    break;
255
256                case BACK :
257                    this.back( vCommand );
258                    break;
259
260                case TEST :
261                    this.test(vCommand);
262                    break;
263
264                case TAKE : 
265                    this.take( vCommand );
266                    break;
267
268                case DROP : 
269                    this.drop( vCommand );
270                    break;
271
272                case ITEMS:
273                    this.items( vCommand );
274                    break;
275
276                case STONE:
277                    this.stone( vCommand );
278                    break;
279                case ALEA:
280                    this.alea( vCommand );
281                    break;
282                    
283                default               :
284                    this.aGui.println ( "Looks like the dev forgot to implement that command but yet whitelisted it." );
285                }
286
287    } // interpretCommand()

290    /**
291     * procedure appellee par la commande alea
292     * @param pCommand la commande entree
293     */
294    public void alea(final Command pCommand)
295    {
296        Room vRoom = this.aPlayer.getLocation();
297
298        if ( vRoom.getClass().equals(TransporterRoom.class ))
299            {
300                TransporterRoom vTRoom = (TransporterRoom) vRoom;
301                if (pCommand.hasSecondWord() )
302                    {
303                        String vWord = pCommand.getSecondWord();
304                        if (vTRoom.doesExist(vWord) )
305                            {
306                                vTRoom.setNext(vWord);
307                                this.aGui.println("Destiny have been altered, you will arrive to "+vWord);
308                            }
309                        else
310                            this.aGui.println("This isn't an available room");
311                    }
312                else
313                    {
314                        vTRoom.setNext(null);
315                        this.aGui.println("Destiny have been altered, you will arrive ... somewhere");
316                    }
317            }
318        else
319            this.aGui.println("You are not in a Transporter Room");
320    } //alea()

324    /**
325     * procedure appellee par la commande stone
326     * @param pCommand la commande entree
327     */
328    public void stone(final Command pCommand)
329    {
330        Beamer vStone = (Beamer) this.aPlayer.getItemByName("stone");
331        if ( vStone == null)
332            this.aGui.println("You don't have any stone.");
333            
334        else if ( pCommand.hasSecondWord() )
335            {
336                String vWord = pCommand.getSecondWord();
337                if (vWord.equals("infuse")) {
338                    vStone.charge(this.aPlayer.getLocation());
339                    this.aGui.println("The stone takes the colour of the ambiant light.");
340                }
341                else if (vWord.equals("jump") )
342                    {
343                        if ( vStone.isCharged() )
344                            {
345                                this.aGui.println("You look into the stone, the stone look into you. Your surroundings changed without you noticing it.");
346                                Room vDestination = vStone.discharge();
347                                this.aPlayer.resetStack();
348                                this.changeRoom( vDestination );
349                            }
350                        else
351                            this.aGui.println("Your look into the stone. Your mind get lost in its plain gray color. You suddendly realize you're getting nowhere.");
352                    }
353                else
354                    this.aGui.println("Stop doing that to the stone!");
355            }
356        else
357            this.aGui.println("You put the stone on your head. It has no effect.");
358        
359    } //stone()

380    private void take( final Command pCommand )
381    {
382        if ( pCommand.hasSecondWord() )
383            {
384                String vSec = pCommand.getSecondWord();
385                Room vLocation = this.aPlayer.getLocation();
386                Item vItem  = vLocation.getItemByName( vSec );
387                if (vItem == null)
388                    this.aGui.println( "This isn't something you can take." );
389                else if (this.aPlayer.canCarryItem( vItem) )
390                {
391                    this.aPlayer.addItem( vItem );
392                    vLocation.removeItem( vItem );
393                    this.aGui.println( "You took a " + vSec );
394                }
395                else
396                    this.aGui.println("You cannot carry more weight.");
397            }
398        else
399            {
400                this.aGui.println ( "Take what?" );         }
401    } //take()

489    /**
490     * Procedure changeant la piece courante.
491     * @param pCommand objet Command dont on sait que le premier mot est 'go'
492     */
493    private void goRoom (final Command pCommand)
494    {
495        if ( ! pCommand.hasSecondWord() )
496            {
497                this.aGui.println("Go where?");
498                return;
499            }
500        
501        Room vNextRoom = this.aPlayer.getLocation().getExit ( pCommand.getSecondWord() );
502        if (vNextRoom == null) {
503           this.aGui.println ( "There is no door!" );
504        } else {
505            if ( vNextRoom.isExit(this.aPlayer.getLocation()) )
506                {
507                    this.aPlayer.pushPreviousLocation( vNextRoom );
508                }
509            else
510                {
511                    this.aPlayer.resetStack();
512                }
513            this.changeRoom( vNextRoom );
514            this.addTime();
515        }
516    } //goRoom()

549    /**
550     * Methode appellee par la commande test
551     * @param pCommand: commande entree
552     */
553    private void test( final Command pCommand )
554    {
555        if ( pCommand.hasSecondWord() )
556            {
557                String vName = pCommand.getSecondWord();
558                if (! vName.matches(".*.txt") )
559                    vName += ".txt";
560                
561                Scanner vScript = RessourcesFetcher.openScript( vName );
562                if (! (vScript == null) )
563                    {
564                        while ( vScript.hasNextLine() )
565                            {
566                                this.interpretCommand( vScript.nextLine() );
567                            }
568                    }
569            } else {
570            this.aGui.println ( "Please specify the script to use." );
571        }
572    } //test()
