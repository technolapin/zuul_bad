001import java.util.HashMap;
002
003public class Inventory
004{
005    private HashMap<Item, Integer> aItems;
006    private int aWeight;
007
008    /**
009     * Constructeur de la classe
010     */
011    public Inventory()
012    {
013        this.aItems = new HashMap<Item, Integer>();
014        this.aWeight = 0;
015    }
016
017
018    /**
019     * Cherche un item dans l'inventaire de l'entite via son nom
020     * @param pItemName le nom de l'item cherche
021     * @return un item ou null
022     */
023    public Item getByName( final String pItemName )
024    {
025        for (Item vItem : this.aItems.keySet() )
026            {
027                if ( vItem.getDescription().equals( pItemName ) )
028                    {
029                        return vItem;
030                    }
031            }
032        return null;
033    }
034
035    
036    /**
037     * Verifie la presence d'un item dans l'inventaire
038     * @param pItem l'item cherche
039     * @return un booleen traduisant la presence de l'item
040     */
041    public boolean hasItem( final Item pItem )
042    {
043        return ( this.aItems.get( pItem ) != null );
044    }
045
046    /**
047     * Verifie si l'inventaire est vide
048     * @return un booleen
049     */
050    public boolean isEmpty()
051    {
052        return this.aItems.isEmpty();
053    }
054
055    /**
056     * ajoute un item a l'inventaire
057     * @param pItem l'item a ajouter
058     */
059    public void add( final Item pItem )
060    {
061        if ( this.hasItem(pItem) )
062            this.aItems.put( pItem, this.aItems.get( pItem ) +1 );
063        else
064            this.aItems.put( pItem, 1 );
065        
066        this.aWeight += pItem.getWeight();
067    }
068
069    /**
070     * ajoute des items a l'inventaire
071     * @param pItem l'item a ajouter
072     * @param pNombre le nombre de fois que pItem doit etre ajoute
073     */
074    public void add( final Item pItem, final int pNombre )
075    {
076        for (int vI = 0; vI < pNombre; vI++)
077            this.add( pItem );
078    }
079
080    /**
081     * retire un item de l'inventaire
082     * @param pItem l'item a supprimer
083     */
084    public void remove( final Item pItem )
085    {
086        if (this.aItems.get( pItem ) > 1)
087            this.aItems.put( pItem, this.aItems.get( pItem ) - 1);
088        else
089            this.aItems.remove( pItem );
090        
091        this.aWeight -= pItem.getWeight();
092    }
093
094    /**
095     * Retourne une chaine decrivant le contenu de l'inventaire
096     * @return une chaine decrivant le contenu de l'inventaire
097     */
098    public String toString()
099    {
100        if ( this.isEmpty() )
101            return "";
102        else
103            {
104                
105                String vS = "";
106                for (Item vItem : this.aItems.keySet())
107                    {
108                        Integer vNombre = this.aItems.get( vItem );
109                        vS += ( "\n            " + vNombre.toString() + " " +
110                                vItem.getDescription() +
111                                ( (vNombre > 1) ? "s":"" ) +
112                                "("+
113                                (new Integer(vItem.getWeight() * vNombre)
114                                 ).toString()+")" );
115
116                    }
117                return vS;
118            }
119    }
120
121    /**
122     * retourne le poids total
123     *@return le poids total
124     */
125    public int totalWeight()
126    {
127        int vTotal = 0;
128        for ( Item vItem : this.aItems.keySet() )
129            {
130                vTotal += vItem.getWeight();
131            }
132        return vTotal;
133    }
134       
135}
