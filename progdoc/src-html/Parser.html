001import java.util.StringTokenizer;
002
003/**
004 * classe traitant les entrees de texte
005 */
006public class Parser 
007{
008
009    private CommandWords aCommandWords;  // holds all valid command words
010
011    
012    /**
013     * Create a new Parser.
014     */
015    public Parser() 
016    {
017        this.aCommandWords = new CommandWords();
018    } // Parser()
019
020    
021    /**
022     * Get a new command from the user. The command is read by
023     * parsing the 'inputLine'.
024     * @param pInputLine la String entree
025     * @return une Commande
026     */
027    public Command
028        getCommand( final String pInputLine ) 
029    {
030        String vWord1;
031        String vWord2;
032
033        StringTokenizer tokenizer = new StringTokenizer( pInputLine );
034
035        if ( tokenizer.hasMoreTokens() )
036            vWord1 = tokenizer.nextToken();      // get first word
037        else
038            vWord1 = null;
039
040        if ( tokenizer.hasMoreTokens() )
041            vWord2 = tokenizer.nextToken();      // get second word
042        else
043            vWord2 = null;
044
045        // note: we just ignore the rest of the input line.
046
047        // Now check whether this word is known. If so, create a command
048        // with it. If not, create a "null" command (for unknown command).
049        
050
051        return new Command( this.aCommandWords.getCommand(vWord1), vWord2 );
052    } // getCommand(.)
053
054    
055    /**
056     * Demande la construction d'une String listant les commandes valides et la retourne.
057     * @return Une string listant les commandes autorisees.
058     */
059    public String
060        getCommandList()
061    {
062        return this.aCommandWords.makeCommandList();
063    } //getCommandList()
064
065} // Parser
