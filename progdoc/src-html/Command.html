001/**
002 * Classe definissant le concept de 'commandes'
003 * @author Cl√©ment Chomicki
004 */
005public class Command
006{
007    private CommandWord aCommandWord;
008    private String aSecondWord;
009
010    /**
011     * Constructeur de la classe
012     * @param pCommandW le mot-commande
013     * @param pSecondW  le second mot
014     */
015    public Command (final CommandWord pCommandW, final String pSecondW )
016    {
017        this.aCommandWord = pCommandW;
018        this.aSecondWord = pSecondW;
019    }
020
021    /**
022     * Accesseur du mot-commande
023     * @return le mot-commande
024     */
025    public CommandWord getCommandWord () { return this.aCommandWord; }
026    /**
027     * Accesseur du second mot
028     * @return le second mot
029     */
030    public String getSecondWord ()  { return this.aSecondWord;  }
031
032    /**
033     * Fonction booleene verifiant si la commande presente un second mot
034     * @return true si et seulement si le second mot n'est pas null.
035     */
036    public boolean hasSecondWord ()
037    {
038        return this.aSecondWord != null;
039    }
040    
041    /**
042     * Fonction booleene verifiant si la commande presente un mot-commande
043     * @return true si et seulement si le mot commande est null.
044     */
045    public boolean isUnknown ()
046    {
047        return this.aCommandWord == CommandWord.UNKNOWN;
048    }
049    
050} // Command
