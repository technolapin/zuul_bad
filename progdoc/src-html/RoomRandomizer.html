001import java.util.HashMap;
002import java.util.ArrayList;
003import java.util.Random;
004
005/**
006 * classe permettant d'obtenir des Room aléatoires
007 * @author Clément Chomicki
008 */
009public class RoomRandomizer
010{
011
012    private HashMap<String, Room> aRooms;
013    private String aNext;
014    private Random aRNG;
015
016    
017    /**
018     * Constructeur
019     * @param pMap le HashMap des Rooms dont disposera le RoomRandomizer
020     */
021    public RoomRandomizer(final HashMap<String, Room> pMap)
022    {
023        this.aRooms = pMap;
024        this.aRNG = new Random();
025    } //RoomRandomizer()
026
027    
028    /**
029     * teste si le mot designe une Room de l'objet
030     * @param pWord le mot a teste
031     * @return un boolean
032     */
033    public boolean doesExist(final String pWord)
034    {
035        return this.aRooms.containsKey(pWord);
036    } //doesExist()
037
038    
039    /**
040     * utile pour la commande alea
041     * @param pString la String designant la prochaine Room tiree
042     */
043    public void setNext(final String pString)
044    {
045        this.aNext = pString;
046    } //setNext()
047
048
049    /**
050     * Retourne la Room correspondante a aNext ou une Room aleatoire
051     * @return une Room parmis celles possibles
052     */
053    public Room getRandomRoom()
054    {
055        if (this.aNext != null)
056            {
057                String vNext = this.aNext;
058                this.aNext = null;
059                return this.aRooms.get(vNext);
060            }
061        else
062            {
063                ArrayList<Room> vList = new ArrayList<Room>(this.aRooms.values());
064                int vI = this.aRNG.nextInt(vList.size());
065                return vList.get(vI);
066            }
067    } //getRandomRoom()
068    
069}
