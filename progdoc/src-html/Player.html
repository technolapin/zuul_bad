001import java.util.Stack;
002
003
004public class Player extends Entity
005{
006    private Room        aLocation;
007    private Stack<Room> aPreviousRooms;
008    private int         aMaxWeight;
009
010    /**
011     * Constructeur de la classe
012     * @param pN le nom de l'entite
013     */
014    public Player( final String pN )
015    {
016        super( pN );
017        this.aPreviousRooms = new Stack<Room>();
018    }
019
020
021
022
023    /**
024     * Accesseur de aLocation
025     * @return la location
026     */
027    public Room getLocation()
028    {
029        return this.aLocation;
030    }
031
032    /**
033     * retire la derniere piece visitee de la pile et la retourne
034     * @return la derniere piece visitee
035     */
036    public Room goToPreviousLocation()
037    {
038        return this.aPreviousRooms.pop();
039    }
040
041    /**
042     * Ajoute une piece a la pile de l'historique des pieces
043     * @param pNextRoom la piece a empiler sur la pile des pieces deja visitees
044     */
045    public void pushPreviousLocation( final Room pNextRoom )
046    {
047        this.aPreviousRooms.push( this.getLocation() );
048    }
049
050    /**
051     * deplace le Joueur a un endroit donne
052     * @param pNextRoom la piece o√π l'on deplace le joueur
053     */
054    public void setLocation( final Room pNextRoom )
055    {
056        this.aLocation = pNextRoom;
057    } // setLocation()
058
059    /**
060     * setter de aMaxWeight
061     * @param pMW nouveau poids maximal
062     */
063    public void setMaxWeight( final int pMW)
064    {
065        this.aMaxWeight = pMW;
066    }
067
068    /**
069     * accesseur de aMaxWeight
070     * @return le poids maximal
071     */
072    public int getMaxWeight()
073    {
074        return this.aMaxWeight;
075    }
076
077    /**
078     * teste si la pile est vide
079     * @return un booleen
080     */
081    public boolean hasNoPreviousLocations()
082    {
083        return this.aPreviousRooms.isEmpty();
084    }
085
086    /**
087     * Cree la description de l'inventaire du joueur
088     * @return la description de l'inventaire du joueur
089     */
090    public String makeItemsDescription()
091    {
092        String vS = super.inventoryString();
093        if (vS.equals(""))
094            return "You own nothing.\nTotal: 0 / "+this.getMaxWeight();
095        else
096            return (
097                    "You have " + vS + "\nTotal: " +
098                    super.getCarriedWeight() +
099                    " / " + this.getMaxWeight()
100                    );
101    }
102
103    public boolean canCarryItem( final Item pItem )
104    {
105        return super.getCarriedWeight() + pItem.getWeight() <= this.getMaxWeight();
106    }
107}
