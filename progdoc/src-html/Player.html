001import java.util.Stack;
002
003/**
004 * la classe du joueur
005 * @author Clément Chomicki
006 */
007public class Player extends Entity
008{
009    private Room        aLocation;
010    private Stack<Room> aPreviousRooms;
011    private int         aMaxWeight;
012
013    
014    /**
015     * Constructeur de la classe
016     * @param pN le nom de l'entite
017     */
018    public Player( final String pN )
019    {
020        super( pN );
021        this.aPreviousRooms = new Stack<Room>();
022    }
023
024    
025    /**
026     * vide la stack
027     */
028    public void resetStack()
029    {
030        this.aPreviousRooms = new Stack<Room>();
031    }
032
033
034    /**
035     * Accesseur de aLocation
036     * @return la location
037     */
038    public Room getLocation()
039    {
040        return this.aLocation;
041    }
042
043    
044    /**
045     * retire la derniere piece visitee de la pile et la retourne
046     * @return la derniere piece visitee
047     */
048    public Room goToPreviousLocation()
049    {
050        return this.aPreviousRooms.pop();
051    }
052
053    
054    /**
055     * Ajoute une piece a la pile de l'historique des pieces
056     * @param pNextRoom la piece a empiler sur la pile des pieces deja visitees
057     */
058    public void pushPreviousLocation( final Room pNextRoom )
059    {
060        this.aPreviousRooms.push( this.getLocation() );
061    }
062
063    
064    /**
065     * deplace le Joueur a un endroit donne
066     * @param pNextRoom la piece où l'on deplace le joueur
067     */
068    public void setLocation( final Room pNextRoom )
069    {
070        this.aLocation = pNextRoom;
071    } // setLocation()
072
073    
074    /**
075     * setter de aMaxWeight
076     * @param pMW nouveau poids maximal
077     */
078    public void setMaxWeight( final int pMW)
079    {
080        this.aMaxWeight = pMW;
081    }
082
083    
084    /**
085     * accesseur de aMaxWeight
086     * @return le poids maximal
087     */
088    public int getMaxWeight()
089    {
090        return this.aMaxWeight;
091    }
092
093    
094    /**
095     * teste si la pile est vide
096     * @return un booleen
097     */
098    public boolean hasNoPreviousLocations()
099    {
100        return this.aPreviousRooms.isEmpty();
101    }
102
103    
104    /**
105     * Cree la description de l'inventaire du joueur
106     * @return la description de l'inventaire du joueur
107     */
108    public String makeItemsDescription()
109    {
110        String vS = super.inventoryString();
111        if (vS.equals(""))
112            return "You own nothing.\nTotal: 0 / "+this.getMaxWeight();
113        else
114            return (
115                    "You have " + vS + "\nTotal: " +
116                    super.getCarriedWeight() +
117                    " / " + this.getMaxWeight()
118                    );
119    }
120
121    
122    /**
123     * determine si un item peut etre porte en fonction de son poids
124     * @param pItem un Item
125     * @return true si l'item est assez leger pour etre porte
126     */
127    public boolean canCarryItem( final Item pItem )
128    {
129        return super.getCarriedWeight() + pItem.getWeight() <= this.getMaxWeight();
130    }
131
132
133    
134}
