012        this.aValidCommands = new HashMap<String, CommandWord>();

014            {

016                    {
017                        for (Language vLang : Language.values())

037
038    /**
039     * Verifie si une String donnee fait partie des commandes valides. 
040     * @param pString la String a tester
041     * @return true si pString est une comande valide, false sinon
042     */
043    public boolean isCommand( final String pString )
044    {
045        return aValidCommands.containsKey(pString);
046    }
047
048    /**
049     * Methode construisant et retournant les commandes autorisees dans une String.
050     * @return Une string listant les commandes autorisees.
051     */
052    public String makeCommandList ()
053    {
054        String vS = ""; 
055        for (String vCmd : aValidCommands.keySet())
056            {
057                vS += vCmd + " ";
058            }
059        return vS;
060    }
061
062    public CommandWord getCommand(final String pWord)
063    {
064        if ( this.isCommand(pWord) )
065            return this.aValidCommands.get(pWord);
066        else
067            return CommandWord.UNKNOWN;
068    }
